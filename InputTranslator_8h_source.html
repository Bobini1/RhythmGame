64template<>
65struct std::hash<input::Key>
66{
67 std::size_t operator()(const input::Key& s) const noexcept
68 {
69 auto gp = s.gamepad.canConvert<input::Gamepad>()
70 ? std::optional{ s.gamepad.value<input::Gamepad>() }
71 : std::nullopt;
72 return std::hash<std::optional<input::Gamepad>>{}(gp) ^
73 std::hash<int>{}(s.code) ^
74 std::hash<int>{}(static_cast<int>(s.device)) ^
75 std::hash<int>{}(static_cast<int>(s.direction));
76 }
