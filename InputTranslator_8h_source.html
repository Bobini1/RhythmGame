63template<>
64struct std::hash<input::Key>
65{
66 std::size_t operator()(const input::Key& s) const noexcept
67 {
68 auto gp = s.gamepad.canConvert<input::Gamepad>()
69 ? std::optional{ s.gamepad.value<input::Gamepad>() }
70 : std::nullopt;
71 return std::hash<std::optional<input::Gamepad>>{}(gp) ^
72 std::hash<int>{}(s.code) ^
73 std::hash<int>{}(static_cast<int>(s.device)) ^
74 std::hash<int>{}(static_cast<int>(s.direction));
75 }
