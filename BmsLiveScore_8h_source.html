20class JudgementCounts final : public QAbstractListModel
21{
22 Q_OBJECT
23
24 QList<int> judgementCounts =
25 QList<int>(magic_enum::enum_count<Judgement>());
26
27 public:
28 explicit JudgementCounts(QObject* parent = nullptr)
29 : QAbstractListModel(parent)
30 {
31 }
32
33 auto rowCount(const QModelIndex& parent = QModelIndex()) const
34 -> int override
35 {
36 return judgementCounts.size();
37 }
38
39 enum Roles
40 {
41 JudgementRole = Qt::UserRole + 1,
42 CountRole
43 };
44
45 auto roleNames() const -> QHash<int, QByteArray> override
46 {
47 return { { JudgementRole, "judgement" }, { CountRole, "count" } };
48 }
49
50 auto data(const QModelIndex& index, int role) const -> QVariant override
51 {
52 if (!index.isValid() || index.row() >= judgementCounts.size()) {
53 return {};
54 }
55 if (role == JudgementRole) {
56 return QVariant::fromValue(static_cast<Judgement>(index.row()));
57 }
58 if (role == CountRole) {
59 return judgementCounts[index.row()];
60 }
61
62 return {};
63 }
64
65 void addJudgement(Judgement judgement)
66 {
67 auto index = magic_enum::enum_integer(judgement);
68 judgementCounts[index]++;
69 emit dataChanged(
70 createIndex(index, 0), createIndex(index, 0), { CountRole });
71 }
72
73 auto getJudgementCounts() const -> const QList<int>&
74 {
75 return judgementCounts;
76 }
77};
